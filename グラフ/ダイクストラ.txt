priority_queue<pll,vpll,greater<pll>> pq;
vll dist(n,INF);
dist[s] = 0;
pq.emplace(dist[s],s);
while(!pq.empty()){
  auto [len,now] = pq.top();
  pq.pop();
  if(dist[now] < len)continue;
  for(auto&[next,movecost]:g[now]){
    if(dist[next] > dist[now] + movecost){
      dist[next] = dist[now] + movecost;
      pq.emplace(dist[next],next);
    }
  }
}
